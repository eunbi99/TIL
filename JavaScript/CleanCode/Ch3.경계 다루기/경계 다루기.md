### 📌 min - max

```
function genRandomNumber(min,max) { 
    return Math.floor(Math.random() * (max - min + 1)) + min;
}

// 상수 
const MIN_NUMBER = 1;
const MAX_NUMBER = 45;

genRandomNumber(MIN_NUMBER,MAX_NUMBER);
```
